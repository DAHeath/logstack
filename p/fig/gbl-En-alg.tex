\begin{algorithm}[H]
  \Fn{$\gEn(e,\cirinp)$}{%
    $\gcirinp \gets \emptystring$\\
    \For{$i \in 0..n\minus 1$}{%
      $\left(X^0, X^1\right) \gets e[i]$\\
      \lIf{$\cirinp[i] = 0$}{$\gcirinp[i] \gets X^0$}
      \lElse{$\gcirinp[i] \gets X^1$}
    }
    \Return{$\gcirinp$}
  }
\end{algorithm}
